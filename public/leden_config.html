// =============================== // CLICK HANDLERS LEDEN-ADMIN // =============================== function resetContainer() { const el = getContainer(); el.className = ""; // verwijder ALLE styles el.style = ""; // verwijder inline styles el.innerHTML = ""; // leegmaken return el; } document.addEventListener('click', async (e) => { // UPDATE if (e.target.classList.contains('changeMember')) { const naam = e.target.dataset.naam || ''; const ok = await showConfirm(`${naam} Wijzigen ❓`); if (!ok) return; const formId = e.target.dataset.formid; const form = document.getElementById(formId); if (form) form.submit(); } // DELETE if (e.target.classList.contains('deleteMember')) { const naam = e.target.dataset.naam; const email = e.target.dataset.email; const ok = await showConfirm(`${naam} verwijderen?`); if (!ok) return; // Vul het verborgen formulier document.getElementById('deleteNaam').value = naam; document.getElementById('deleteEmail').value = email; // Verstuur het formulier document.getElementById('deleteForm').submit(); } }); // EXPORT document.getElementById('exportBtn').onclick = () => { const link = document.createElement('a'); link.href = "leden_export.php"; link.download = "leden.xlsx"; link.click(); setTimeout(() => { showAlert("success", "xlsx is aangemaakt ✔️"); // SERVER-SIDE MESSAGE }, 500); };
